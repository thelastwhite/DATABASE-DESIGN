CREATE OR REPLACE EDITIONABLE TRIGGER  "BI_ADDRESS" 
  before insert on "ADDRESS"               
  for each row  
begin   
  if :NEW."ADDRESS_NO" is null then 
    select "ADDRESS_SEQ".nextval into :NEW."ADDRESS_NO" from sys.dual;
